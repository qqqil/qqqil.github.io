排序

  1. 选择排序 

    void selection_sorting(vector<int> &nums){
 int sz = nums.size();
 for(int i=0;i<sz;i++){
     int min = i;
     for(int j =i+1;j<sz;j++){
         if(nums[j] <nums[i]){
             min =j;
         }
     }
     //exch
     int tmp = nums[i];
     nums[i] = nums[min];
     nums[min] = tmp;
 }
}

    void exch(vector<int> &nums,int i,int j){
 int tmp = nums[i];
 nums[i] = nums[j];
 nums[j] = tmp;
}
bool mless(int val1,int val2){
 return val1 <val2;
}
void insert_sorting(vector<int> &nums){
 size_t  sz = nums.size();
 for(int i=0;i<sz;i++){
   for(int j =i;j>0;j--){
       if(mless(nums[j],nums[j-1])){
           exch(nums,j,j-1);
       }
   }
 }
}
